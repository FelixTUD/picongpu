38  HINLINE float3_X laserLongitudinal( uint32_t currentStep, float_X& phase )
39  {
40  /* initialize the laser not in the first cell is equal to a negative shift
41  * in time
42  */
43  constexpr float_X laserTimeShift = laser::initPlaneY * CELL_HEIGHT / SPEED_OF_LIGHT;
44  const float_64 runTime = DELTA_T * currentStep - laserTimeShift;
45  const float_64 f = SPEED_OF_LIGHT / WAVE_LENGTH;
46 
47  /* calculate focus position relative to the laser initialization plane */
48  constexpr float_X focusPos = FOCUS_POS - laser::initPlaneY * CELL_HEIGHT;
49 
50  float3_X elong(float3_X::create(0.0));
51 
52  // a symmetric pulse will be initialized at position z=0 for
53  // a time of PULSE_INIT * PULSE_LENGTH = INIT_TIME.
54  // we shift the complete pulse for the half of this time to start with
55  // the front of the laser pulse.
56  const float_64 mue = 0.5 * INIT_TIME;
57 
58  //rayleigh length (in y-direction)
59  const float_64 y_R = PI * W0 * W0 / WAVE_LENGTH;
60  //gaussian beam waist in the nearfield: w_y(y=0) == W0
61  const float_64 w_y = W0 * sqrt( 1.0 + ( focusPos / y_R )*( focusPos / y_R ) );
62 
63  float_64 envelope = float_64( AMPLITUDE );
64  if( simDim == DIM2 )
65  envelope *= math::sqrt( float_64( W0 ) / w_y );
66  else if( simDim == DIM3 )
67  envelope *= float_64( W0 ) / w_y;
68  /* no 1D representation/implementation */
69 
70  if( Polarisation == LINEAR_X )
71  {
72  elong.x() = float_X( envelope );
73  }
74  else if( Polarisation == LINEAR_Z )
75  {
76  elong.z() = float_X( envelope );
77  }
78  else if( Polarisation == CIRCULAR )
79  {
80  elong.x() = float_X( envelope / sqrt(2.0) );
81  elong.z() = float_X( envelope / sqrt(2.0) );
82  }
83 
84  phase = float_X(2.0) * float_X(PI ) * float_X(f ) * ( runTime - float_X(mue ) - focusPos / SPEED_OF_LIGHT ) + LASER_PHASE;
85 
86  return elong;
87  }

97  HDINLINE float3_X laserTransversal( float3_X elong, float_X phase, const float_X posX, const float_X posZ )
98  {
99  //const float_X modMue = float_X(PI) * float_X(SPEED_OF_LIGHT / WAVE_LENGTH) * INIT_TIME;
100  const float_X f = SPEED_OF_LIGHT / WAVE_LENGTH;
101  /* calculate focus position relative to the laser initialization plane */
102  constexpr float_X focusPos = FOCUS_POS - laser::initPlaneY * CELL_HEIGHT;
103 
104  const float_X timeShift = phase / (float_X(2.0) * float_X(PI) * float_X(f)) + focusPos / SPEED_OF_LIGHT;
105  const float_X local_tilt_x = TILT_X;
106  const float_X spaceShift = SPEED_OF_LIGHT * algorithms::math::tan(local_tilt_x) * timeShift / CELL_HEIGHT;
107  const float_X r2 = (posX + spaceShift) * (posX + spaceShift) + posZ * posZ;
108 
109 
110  // pure gaussian
111  //const float_X r2 = posX * posX + posZ * posZ;
112 
113  //rayleigh length (in y-direction)
114  const float_X y_R = float_X( PI ) * W0 * W0 / WAVE_LENGTH;
115 
116  // the radius of curvature of the beam's wavefronts
117  const float_X R_y = -focusPos * ( float_X(1.0) + ( y_R / focusPos )*( y_R / focusPos ) );
118 
119  //beam waist in the near field: w_y(y=0) == W0
120  const float_X w_y = W0 * algorithms::math::sqrt( float_X(1.0) + ( focusPos / y_R )*( focusPos / y_R ) );
122  const float_X xi_y = algorithms::math::atan( -focusPos / y_R );
123 
124  if( Polarisation == LINEAR_X || Polarisation == LINEAR_Z )
125  {
126  elong *= math::exp( -r2 / w_y / w_y ) * math::cos( float_X(2.0) * float_X( PI ) / WAVE_LENGTH * focusPos - float_X(2.0) * float_X( PI ) / WAVE_LENGTH * r2 / float_X(2.0) / R_y + xi_y + phase )
127  * math::exp( -( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
128  *( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
129  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( float_X(2.0) * PULSE_LENGTH ) / ( float_X(2.0) * PULSE_LENGTH ) );
130  }
131  else if( Polarisation == CIRCULAR )
132  {
133  elong.x() *= math::exp( -r2 / w_y / w_y ) * math::cos( float_X(2.0) * float_X( PI ) / WAVE_LENGTH * focusPos - float_X(2.0) * float_X( PI ) / WAVE_LENGTH * r2 / float_X(2.0) / R_y + xi_y + phase )
134  * math::exp( -( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
135  *( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
136  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( float_X(2.0) * PULSE_LENGTH ) / ( float_X(2.0) * PULSE_LENGTH ) );
137  phase += float_X( PI / 2.0 );
138  elong.z() *= math::exp( -r2 / w_y / w_y ) * math::cos( float_X(2.0) * float_X( PI ) / WAVE_LENGTH * focusPos - float_X(2.0) * float_X( PI ) / WAVE_LENGTH * r2 / float_X(2.0) / R_y + xi_y + phase )
139  * math::exp( -( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
140  *( r2 / float_X(2.0) / R_y - focusPos - phase / float_X(2.0) / float_X( PI ) * WAVE_LENGTH )
141  / SPEED_OF_LIGHT / SPEED_OF_LIGHT / ( float_X(2.0) * PULSE_LENGTH ) / ( float_X(2.0) * PULSE_LENGTH ) );
142  phase -= float_X( PI / 2.0 );
143  }
144 
145  return elong;
146  }
