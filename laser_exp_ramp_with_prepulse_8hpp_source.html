80  HINLINE float_X
81  get_envelope(float_X runTime)
82  {
83  float_X env = 0.0;
84  const bool before_preupramp = runTime < time_start_init;
85  const bool before_start = runTime < TIME_1;
86  const bool before_peakpulse = runTime < endUpramp;
87  const bool during_first_exp = ( TIME_1 < runTime ) &&
88  ( runTime < TIME_2 );
89  const bool after_peakpulse = startDownramp <= runTime;
90 
91  if ( before_preupramp )
92  env = 0.;
93  else if ( before_start )
94  {
95  env = AMP_1 * gauss( runTime - TIME_1 );
96  }
97  else if ( before_peakpulse )
98  {
99  const float_X ramp_when_peakpulse = extrapolate_expo(
100  TIME_2,
101  AMP_2,
102  TIME_3,
103  AMP_3,
104  endUpramp
105  ) / AMPLITUDE;
106 
107  if ( ramp_when_peakpulse > 0.5 )
108  {
109  log< picLog::PHYSICS >(
110  "Attention, the intensities of the laser upramp are very large! "
111  "The extrapolation of the last exponential to the time of "
112  "the peakpulse gives more than half of the amplitude of "
113  "the peak Gaussian. This is not a Gaussian at all anymore, "
114  "and physically very unplausible, check the params for misunderstandings!"
115  );
116  }
117 
118  env += AMPLITUDE * ( float_X( 1. ) - ramp_when_peakpulse ) *
119  gauss( runTime - endUpramp );
120  env += AMP_PREPULSE * gauss( runTime - TIME_PREPULSE );
121  if ( during_first_exp )
122  env += extrapolate_expo(
123  TIME_1,
124  AMP_1,
125  TIME_2,
126  AMP_2,
127  runTime
128  );
129  else
130  env += extrapolate_expo(
131  TIME_2,
132  AMP_2,
133  TIME_3,
134  AMP_3,
135  runTime
136  );
137  }
138  else if ( !after_peakpulse )
139  env = AMPLITUDE;
140  else // after startDownramp
141  env = AMPLITUDE * gauss( runTime - startDownramp );
142  return env;
143  }

145  HINLINE float3_X laserLongitudinal(uint32_t currentStep, float_X& phase)
146  {
147  float_X envelope;
148  float3_X elong( float3_X::create( 0.0 ) );
149 
150  // a symmetric pulse will be initialized at position z=0 for
151  // a time of RAMP_INIT * PULSE_LENGTH + LASER_NOFOCUS_CONSTANT = INIT_TIME.
152  // we shift the complete pulse for the half of this time to start with
153  // the front of the laser pulse.
154 
155  /* initialize the laser not in the first cell is equal to a negative shift
156  * in time
157  */
158  const float_64 runTime = time_start_init - laserTimeShift +
159  DELTA_T * currentStep;
160 
161  phase += float_X( w * runTime ) + LASER_PHASE;
162 
163  envelope = get_envelope( runTime );
164 
165  if( Polarisation == LINEAR_X )
166  {
167  elong.x() = float_X( envelope * ( math::sin( phase ) ) );
168  }
169  else if( Polarisation == LINEAR_Z )
170  {
171  elong.z() = float_X( envelope * ( math::sin( phase ) ) );
172  }
173  else if( Polarisation == CIRCULAR )
174  {
175  elong.x() = float_X( envelope / sqrt( 2.0 ) * ( math::sin( phase ) ) );
176  elong.z() = float_X( envelope / sqrt( 2.0 ) * ( math::cos( phase ) ) );
177  }
178 
179  return elong;
180  }
